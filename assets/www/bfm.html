Info
Show content of file
cat [file_name]
Show file content from the top
head -n [no_of_lines] [file_name]
Show file content from the bottom
tail -n [no_of_lines] [file_name]
Create Directory/Folder
mkdir [directory_name]
Copy file
cp [source_file] [destination_file]
Copy folder
cp -R [source_dir] [destination_dir]
Move file
mv [source_file] [destination_file]
Delete file
rm [file_name]
Force deletion of file
rm -f [file_name]
Delete Folder
rm -r [dir_name]
Create zero byte(empty) file
touch [file_name]
Link file(Physical)
ln [file1] [file2]
Link file(Symbolic)
ln -s [file1] [file2]
